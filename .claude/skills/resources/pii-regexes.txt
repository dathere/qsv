# PII regex patterns for use with qsv searchset
# Canonical source: docs/cookbook/searchset/pii-regexes.txt
(?x)(?:\b\d{3}[\s-]?\d{2}[\s-]?\d{4}\b) #SSN
(?x)\b5[1-5]\d{2}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b # Mastercard
(?x)\b4\d{3}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b # Visa
(?x)3[47][0-9]{13} # American Express
(?x)[A-Z]{2}\d{2}[A-Z0-9]{4}\d{7}([A-Z\d]?){0,16} # IBAN
(?x)[\w\.=-]+@[\w\.-]+\.[\w]{2,63} # Email
(?x)(?:\+?1)[\s\.]?(([2-9]\d{2})|\(([2-9]\d{2})\))[\s\.\-]?([2-9]\d{2})[\s\.\-]?(\d{4}) #Phone
